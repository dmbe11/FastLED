379 FASTLED_FORCE_INLINE void maximizeBrightness( uint8_t limit = 255 ) {
380 uint8_t max = red;
381 if( green > max) max = green;
382 if( blue > max) max = blue;
383
384 // stop div/0 when color is black
385 if(max > 0) {
386 uint16_t factor = ((uint16_t)(limit) * 256) / max;
387 red = (red * factor) / 256;
388 green = (green * factor) / 256;
389 blue = (blue * factor) / 256;
390 }

434 FASTLED_FORCE_INLINE void setParity( uint8_t parity)
435 {
436 uint8_t curparity = getParity();
437
438 if( parity == curparity) return;
439
440 if( parity ) {
441 // going 'up'
442 if( (b > 0) && (b < 255)) {
443 if( r == g && g == b) {
444 ++r;
445 ++g;
446 }
447 ++b;
448 } else if( (r > 0) && (r < 255)) {
449 ++r;
450 } else if( (g > 0) && (g < 255)) {
451 ++g;
452 } else {
453 if( r == g && g == b) {
454 r ^= 0x01;
455 g ^= 0x01;
456 }
457 b ^= 0x01;
458 }
459 } else {
460 // going 'down'
461 if( b > 1) {
462 if( r == g && g == b) {
463 --r;
464 --g;
465 }
466 --b;
467 } else if( g > 1) {
468 --g;
469 } else if( r > 1) {
470 --r;
471 } else {
472 if( r == g && g == b) {
473 r ^= 0x01;
474 g ^= 0x01;
475 }
476 b ^= 0x01;
477 }
478 }
