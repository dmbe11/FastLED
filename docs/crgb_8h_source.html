376 FASTLED_FORCE_INLINE void maximizeBrightness( uint8_t limit = 255 ) {
377 uint8_t max = red;
378 if( green > max) max = green;
379 if( blue > max) max = blue;
380
381 // stop div/0 when color is black
382 if(max > 0) {
383 uint16_t factor = ((uint16_t)(limit) * 256) / max;
384 red = (red * factor) / 256;
385 green = (green * factor) / 256;
386 blue = (blue * factor) / 256;
387 }

431 FASTLED_FORCE_INLINE void setParity( uint8_t parity)
432 {
433 uint8_t curparity = getParity();
434
435 if( parity == curparity) return;
436
437 if( parity ) {
438 // going 'up'
439 if( (b > 0) && (b < 255)) {
440 if( r == g && g == b) {
441 ++r;
442 ++g;
443 }
444 ++b;
445 } else if( (r > 0) && (r < 255)) {
446 ++r;
447 } else if( (g > 0) && (g < 255)) {
448 ++g;
449 } else {
450 if( r == g && g == b) {
451 r ^= 0x01;
452 g ^= 0x01;
453 }
454 b ^= 0x01;
455 }
456 } else {
457 // going 'down'
458 if( b > 1) {
459 if( r == g && g == b) {
460 --r;
461 --g;
462 }
463 --b;
464 } else if( g > 1) {
465 --g;
466 } else if( r > 1) {
467 --r;
468 } else {
469 if( r == g && g == b) {
470 r ^= 0x01;
471 g ^= 0x01;
472 }
473 b ^= 0x01;
474 }
475 }
