583LIB8STATIC uint8_t ease8InOutQuad( uint8_t i)
584{
585 uint8_t j = i;
586 if( j & 0x80 ) {
587 j = 255 - j;
588 }
589 uint8_t jj = scale8( j, j);
590 uint8_t jj2 = jj << 1;

593 }

598// This AVR asm version of ease8InOutQuad preserves one more
599// low-bit of precision than the C version, and is also slightly
600// smaller and faster.
601LIB8STATIC uint8_t ease8InOutQuad(uint8_t val) {
602 uint8_t j=val;
603 asm volatile (
604 "sbrc %[val], 7 \n"
605 "com %[j] \n"
606 "mul %[j], %[j] \n"
607 "add r0, %[j] \n"
608 "ldi %[j], 0 \n"
609 "adc %[j], r1 \n"
610 "lsl r0 \n" // carry = high bit of low byte of mul product
611 "rol %[j] \n" // j = (j * 2) + carry // preserve add'l bit of precision
612 "sbrc %[val], 7 \n"
613 "com %[j] \n"
614 "clr __zero_reg__ \n"
615 : [j] "+&a" (j)
616 : [val] "a" (val)
617 : "r0", "r1"
618 );
619 return j;

645LIB8STATIC fract8 ease8InOutCubic( fract8 i)
646{
647 uint8_t ii = scale8_LEAVING_R1_DIRTY( i, i);
648 uint8_t iii = scale8_LEAVING_R1_DIRTY( ii, i);
649
650 uint16_t r1 = (3 * (uint16_t)(ii)) - ( 2 * (uint16_t)(iii));
651
652 /* the code generated for the above *'s automatically
653 cleans up R1, so there's no need to explicitily call
654 cleanup_R1(); */
655
656 uint8_t result = r1;
657
658 // if we got "256", return 255:
659 if( r1 & 0x100 ) {
660 result = 255;
661 }
662 return result;
663}

674LIB8STATIC fract8 ease8InOutApprox( fract8 i)
675{
676 if( i < 64) {
677 // start with slope 0.5
678 i /= 2;
679 } else if( i > (255 - 64)) {
680 // end with slope 0.5
681 i = 255 - i;
682 i /= 2;
683 i = 255 - i;
684 } else {
685 // in the middle, use slope 192/128 = 1.5
686 i -= 64;
687 i += (i / 2);
688 i += 32;
689 }
690
691 return i;
692}

951LIB8STATIC uint16_t beat88( accum88 beats_per_minute_88, uint32_t timebase = 0)
952{
953 // BPM is 'beats per minute', or 'beats per 60000ms'.
954 // To avoid using the (slower) division operator, we
955 // want to convert 'beats per 60000ms' to 'beats per 65536ms',
956 // and then use a simple, fast bit-shift to divide by 65536.
957 //
958 // The ratio 65536:60000 is 279.620266667:256; we'll call it 280:256.
959 // The conversion is accurate to about 0.05%, more or less,
960 // e.g. if you ask for "120 BPM", you'll get about "119.93".
961 return (((GET_MILLIS()) - timebase) * beats_per_minute_88 * 280) >> 16;
962}

967LIB8STATIC uint16_t beat16( accum88 beats_per_minute, uint32_t timebase = 0)
968{
969 // Convert simple 8-bit BPM's to full Q8.8 accum88's if needed
970 if( beats_per_minute < 256) beats_per_minute <<= 8;
971 return beat88(beats_per_minute, timebase);
972}
