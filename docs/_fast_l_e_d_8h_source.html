402 template<ESPIChipsets CHIPSET, uint8_t DATA_PIN, uint8_t CLOCK_PIN>
403 struct ClockedChipsetHelper {
404 // Default implementation, will be specialized for supported chipsets
405 static const bool IS_VALID = false;
406 };
407
408 // Macro to define a mapping from the ESPIChipeset enum to the controller class
409 // in it's various template configurations.
410 #define _FL_MAP_CLOCKED_CHIPSET(CHIPSET_ENUM, CONTROLLER_CLASS) \
411 template<uint8_t DATA_PIN, uint8_t CLOCK_PIN> \
412 struct ClockedChipsetHelper<CHIPSET_ENUM, DATA_PIN, CLOCK_PIN> { \
413 static const bool IS_VALID = true; \
414 typedef CONTROLLER_CLASS<DATA_PIN, CLOCK_PIN> ControllerType; \
415 /* Controller type with RGB_ORDER specified */ \
416 template<EOrder RGB_ORDER> \
417 struct CONTROLLER_CLASS_WITH_ORDER { \
418 typedef CONTROLLER_CLASS<DATA_PIN, CLOCK_PIN, RGB_ORDER> ControllerType; \
419 }; \
420 /* Controller type with RGB_ORDER and spi frequency specified */ \
421 template<EOrder RGB_ORDER, uint32_t FREQ> \
422 struct CONTROLLER_CLASS_WITH_ORDER_AND_FREQ { \
423 typedef CONTROLLER_CLASS<DATA_PIN, CLOCK_PIN, RGB_ORDER, FREQ> ControllerType; \
424 }; \
425 };
426
427 // Define specializations for each supported CHIPSET
428 _FL_MAP_CLOCKED_CHIPSET(LPD6803, LPD6803Controller)
429 _FL_MAP_CLOCKED_CHIPSET(LPD8806, LPD8806Controller)
430 _FL_MAP_CLOCKED_CHIPSET(WS2801, WS2801Controller)

432 _FL_MAP_CLOCKED_CHIPSET(SM16716, SM16716Controller)
433 _FL_MAP_CLOCKED_CHIPSET(P9813, P9813Controller)
