172 if (!ok) {

178 mEffects.next(mCurrId, &mCurrId, true);

195inline bool FxEngine::draw(uint32_t now, CRGB *finalBuffer) {
196 mTimeWarp.setTimeScale(mTimeBender);
197 mTimeWarp.update(now);
198 uint32_t warpedTime = mTimeWarp.getTime();
199
200 if (mEffects.empty()) {
201 return false;
202 }
203 if (mDurationSet) {
204 FxPtr fx;
205 bool ok = mEffects.get(mCurrId, &fx);
206 if (!ok) {
207 // something went wrong.
208 return false;
209 }
210 mCompositor.startTransition(now, mDuration, fx);
211 mDurationSet = false;
212 }
213 if (!mEffects.empty()) {
214 mCompositor.draw(now, warpedTime, finalBuffer);
215 }
216 return true;
