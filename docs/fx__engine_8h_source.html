167inline FxRef FxEngine::removeFx(int index) {
168 if (!mEffects.has(index)) {
169 return FxRef();
170 }
171
172 FxRef removedFx;
173 bool ok = mEffects.get(index, &removedFx);
174 if (!ok) {
175 return FxRef();
176 }
177
178 if (mCurrId == index) {
179 // If we're removing the current effect, switch to the next one
180 mEffects.next(mCurrId, &mCurrId, true);
181 mDurationSet = true;
182 mDuration = 0; // Instant transition
183 }
184
185 return removedFx;
186}

197inline bool FxEngine::draw(uint32_t now, CRGB *finalBuffer) {
198 mTimeWarp.setTimeScale(mTimeBender);
199 mTimeWarp.update(now);
200 uint32_t warpedTime = mTimeWarp.getTime();
201
202 if (mEffects.empty()) {
203 return false;
204 }
205 if (mDurationSet) {
206 FxRef fx;
207 bool ok = mEffects.get(mCurrId, &fx);
208 if (!ok) {
209 // something went wrong.
210 return false;
211 }
212 mCompositor.startTransition(now, mDuration, fx);
213 mDurationSet = false;
214 }
215 if (!mEffects.empty()) {
216 mCompositor.draw(now, warpedTime, finalBuffer);
217 }
218 return true;
219}
