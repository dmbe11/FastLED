61void setup() {
62 Serial.begin(115200);
63 delay(1000); // sanity delay
64 if (!filesystem.beginSd(CHIP_SELECT_PIN)) {
65 Serial.println("Failed to initialize file system.");
66 }
67 // JSON_SCREEN_MAP
68 FixedMap<Str, ScreenMap, 16> screenMaps;
69 ScreenMap::ParseJson(JSON_SCREEN_MAP, &screenMaps);
70 ScreenMap screenMap = screenMaps["strip1"];
71
72 FastLED.addLeds<LED_TYPE, LED_PIN, COLOR_ORDER>(leds, NUM_LEDS)
73 .setCorrection(TypicalLEDStrip)
74 .setScreenMap(screenMap);
75 FastLED.setBrightness(96);
76 // Note that data/ is a special directory used by our wasm compiler. Any data
77 // is placed in it will be included in the files.json file which the browser will
78 // use to stream the file asynchroniously in during the runtime. For a real sd card
79 // just place all this in the /data/ directory of the SD card to get matching
80 // behavior.
81 video = filesystem.openVideo("data/video.dat", NUM_LEDS, FPS, 2);
82 if (!video) {
83 Serial.println("Failed to instantiate video");
84 }
85}
