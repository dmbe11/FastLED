99bool VideoImpl::draw(uint32_t now, CRGB *leds, uint8_t *alpha) {
100 if (!mStream) {
101 return false;
102 }
103 bool ok = updateBufferIfNecessary(mPrevNow, now);
104 mPrevNow = now;
105 if (!ok) {
106 // paint black
107 memset(leds, 0, mPixelsPerFrame * sizeof(CRGB));
108 if (alpha) {
109 memset(alpha, 0, mPixelsPerFrame);

111 return false;
112 }

117bool VideoImpl::updateBufferIfNecessary(uint32_t prev, uint32_t now) {
118 const bool forward = now >= prev;
119 // At most, update one frame. That way if the user forgets to call draw and
120 // then sends a really old timestamp, we don't update the buffer too much.
121 uint32_t currFrameNumber = 0;
122 uint32_t nextFrameNumber = 0;
123 while (mFrameInterpolator->needsFrame(now, &currFrameNumber, &nextFrameNumber)) {
124 if (mFrameInterpolator->empty()) {
125 // we are missing the first frame
126 FrameRef frame = FrameRef::New(mPixelsPerFrame, false);
127 if (!mStream->readFrame(frame.get())) {
128 if (!mStream->rewind()) {
129 FASTLED_DBG("readFrame (1) failed");
130 return false;
131 }
132 if (!mStream->readFrame(frame.get())) {
133 FASTLED_DBG("readFrame (2) failed");
134 return false;
135 }
136 }
137 uint32_t timestamp = mFrameInterpolator->get_exact_timestamp_ms(currFrameNumber);
138 frame->setFrameNumberAndTime(currFrameNumber, timestamp);
139 mFrameInterpolator->push_front(frame, currFrameNumber, timestamp);
140 continue;
141 }
142
143 uint32_t newest_frame_number = 0;
144 bool has_newest = mFrameInterpolator->get_newest_frame_number(&newest_frame_number);
145 if (!has_newest) {
146 DBG("get_newest_frame_number failed");
147 return false;
148 }
149
150 FrameRef frame;
151 if (full()) {
152 frame = popOldest();
153 } else {
154 // We will continue allocating until the buffer fills up with frames, then we start
155 // recycling.
156 frame = FrameRef::New(mPixelsPerFrame, false);
157 }
158 if (!mStream->readFrame(frame.get())) {
159 if (!mStream->rewind()) {
160 DBG("readFrame (3) failed");
161 return false;
162 }
163 if (!mStream->readFrame(frame.get())) {
164 DBG("readFrame (4) failed");
165 return false;
166 }
167 }
168 uint32_t next_frame = newest_frame_number + 1;
169 uint32_t next_timestamp = mFrameInterpolator->get_exact_timestamp_ms(next_frame);
170 frame->setFrameNumberAndTime(next_frame, next_timestamp);
171 bool ok = mFrameInterpolator->push_front(frame, next_frame, next_timestamp);
172 if (!ok) {
173 DBG("pushNewest failed");
174 return false;
175 }
176 }
177 return true;
178}

265void VideoFx::draw(DrawContext context) {
266 if (!mFrame) {
267 mFrame = FrameRef::New(mXyMap.getTotal(), false);
268 }
269 bool ok = mVideo.draw(context.now, mFrame.get());
270 if (!ok) {
271 mVideo.rewind();
272 ok = mVideo.draw(context.now, mFrame.get());
273 if (!ok) {
274 return; // Can't draw or rewind
275 }
276 }
277 if (!mFrame) {
278 return; // Can't draw without a frame
279 }
280
281 const CRGB *src_pixels = mFrame->rgb();
282 CRGB *dst_pixels = context.leds;
283 size_t dst_pos = 0;
284 for (uint16_t w = 0; w < mXyMap.getWidth(); w++) {
285 for (uint16_t h = 0; h < mXyMap.getHeight(); h++) {
286 const size_t index = mXyMap.mapToIndex(w, h);
287 if (index < mFrame->size()) {
288 dst_pixels[dst_pos++] = src_pixels[index];
289 }
290 }
291 }
