38inline void bilinearExpand(const CRGB *input, CRGB *output, uint16_t inputWidth,
39 uint16_t inputHeight, XYMap xyMap) {
40 uint16_t outputWidth = xyMap.getWidth();
41 uint16_t outputHeight = xyMap.getHeight();
42 const bool wontFit = (outputWidth != xyMap.getWidth() || outputHeight != xyMap.getHeight());
43 // if the input dimensions are not a power of 2 then we can't use the
44 // optimized version.
45 if (wontFit || (inputWidth & (inputWidth - 1)) || (inputHeight & (inputHeight - 1))) {
46 bilinearExpandArbitrary(input, output, inputWidth, inputHeight, xyMap);
47 } else {
48 bilinearExpandPowerOf2(input, output, inputWidth, inputHeight, xyMap);
49 }
50}
