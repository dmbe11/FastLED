107void fill_gradient( T* targetArray,
108 uint16_t startpos, CHSV startcolor,
109 uint16_t endpos, CHSV endcolor,
110 TGradientDirectionCode directionCode = SHORTEST_HUES )
111{
112 // if the points are in the wrong order, straighten them
113 if( endpos < startpos ) {
114 uint16_t t = endpos;
115 CHSV tc = endcolor;
116 endcolor = startcolor;
117 endpos = startpos;
118 startpos = t;
119 startcolor = tc;
120 }
121
122 // If we're fading toward black (val=0) or white (sat=0),
123 // then set the endhue to the starthue.
124 // This lets us ramp smoothly to black or white, regardless
125 // of what 'hue' was set in the endcolor (since it doesn't matter)
126 if( endcolor.value == 0 || endcolor.saturation == 0) {
127 endcolor.hue = startcolor.hue;
128 }
129
130 // Similarly, if we're fading in from black (val=0) or white (sat=0)
131 // then set the starthue to the endhue.
132 // This lets us ramp smoothly up from black or white, regardless
133 // of what 'hue' was set in the startcolor (since it doesn't matter)
134 if( startcolor.value == 0 || startcolor.saturation == 0) {
135 startcolor.hue = endcolor.hue;
136 }
137
138 saccum87 huedistance87;
139 saccum87 satdistance87;
140 saccum87 valdistance87;
141
142 satdistance87 = (endcolor.sat - startcolor.sat) << 7;
143 valdistance87 = (endcolor.val - startcolor.val) << 7;
144
145 uint8_t huedelta8 = endcolor.hue - startcolor.hue;
146
147 if( directionCode == SHORTEST_HUES ) {
148 directionCode = FORWARD_HUES;
149 if( huedelta8 > 127) {
150 directionCode = BACKWARD_HUES;
151 }
152 }
153
154 if( directionCode == LONGEST_HUES ) {
155 directionCode = FORWARD_HUES;
156 if( huedelta8 < 128) {
157 directionCode = BACKWARD_HUES;
158 }
159 }
160
161 if( directionCode == FORWARD_HUES) {
162 huedistance87 = huedelta8 << 7;
163 }
164 else /* directionCode == BACKWARD_HUES */
165 {
166 huedistance87 = (uint8_t)(256 - huedelta8) << 7;
167 huedistance87 = -huedistance87;
168 }
169
170 uint16_t pixeldistance = endpos - startpos;
171 int16_t divisor = pixeldistance ? pixeldistance : 1;
172
173 #if FASTLED_USE_32_BIT_GRADIENT_FILL
174 // Use higher precision 32 bit math for new micros.
175 int32_t huedelta823 = (huedistance87 * 65536) / divisor;
176 int32_t satdelta823 = (satdistance87 * 65536) / divisor;
177 int32_t valdelta823 = (valdistance87 * 65536) / divisor;
178
179 huedelta823 *= 2;
180 satdelta823 *= 2;
181 valdelta823 *= 2;
182 uint32_t hue824 = static_cast<uint32_t>(startcolor.hue) << 24;
183 uint32_t sat824 = static_cast<uint32_t>(startcolor.sat) << 24;
184 uint32_t val824 = static_cast<uint32_t>(startcolor.val) << 24;
185 for( uint16_t i = startpos; i <= endpos; ++i) {
186 targetArray[i] = CHSV( hue824 >> 24, sat824 >> 24, val824 >> 24);
187 hue824 += huedelta823;
188 sat824 += satdelta823;
189 val824 += valdelta823;
190 }
191 #else
192 // Use 8-bit math for older micros.
193 saccum87 huedelta87 = huedistance87 / divisor;
194 saccum87 satdelta87 = satdistance87 / divisor;
195 saccum87 valdelta87 = valdistance87 / divisor;
196
197 huedelta87 *= 2;
198 satdelta87 *= 2;
199 valdelta87 *= 2;
200
201 accum88 hue88 = startcolor.hue << 8;
202 accum88 sat88 = startcolor.sat << 8;
203 accum88 val88 = startcolor.val << 8;
204 for( uint16_t i = startpos; i <= endpos; ++i) {
205 targetArray[i] = CHSV( hue88 >> 8, sat88 >> 8, val88 >> 8);
206 hue88 += huedelta87;
207 sat88 += satdelta87;
208 val88 += valdelta87;
209 }
210 #endif // defined(__AVR__)
211}
