94// Emulution layer to support RGBW leds on RGB controllers. This works by creating
95// a side buffer dedicated for the RGBW data. The RGB data is then converted to RGBW
96// and sent to the delegate controller for rendering as if it were RGB data.
97template <
98 typename CONTROLLER,
99 EOrder RGB_ORDER = GRB> // Default on WS2812>
100class RGBWEmulatedController
101 : public CPixelLEDController<RGB_ORDER, CONTROLLER::LANES_VALUE,
102 CONTROLLER::MASK_VALUE> {
103 public:
104 static const int LANES = CONTROLLER::LANES_VALUE;
105 static const uint32_t MASK = CONTROLLER::MASK_VALUE;
106
107 // The delegated controller must do no reordering.
108 static_assert(RGB == CONTROLLER::RGB_ORDER_VALUE, "The delegated controller MUST NOT do reordering");
109
110 RGBWEmulatedController(const Rgbw& rgbw = RgbwDefault()) {
111 this->setRgbw(rgbw);
112 };
113 ~RGBWEmulatedController() { delete[] mRGBWPixels; }
114
115 virtual void showPixels(PixelController<RGB_ORDER, LANES, MASK> &pixels) {
116 // Ensure buffer is large enough
117 ensureBuffer(pixels.size());
118 // This version sent down to the real controller.
119 PixelController<RGB, LANES, MASK> pixels_device(pixels);
120 pixels_device.mColorAdjustment.premixed = CRGB(255, 255, 255); // No scaling because we do that.
121 #if FASTLED_HD_COLOR_MIXING
122 pixels_device.mColorAdjustment.color = CRGB(255, 255, 255);
123 pixels_device.mColorAdjustment.brightness = 255;
124 #endif
125 pixels_device.mData = reinterpret_cast<uint8_t *>(mRGBWPixels);
126 pixels_device.mLen = mNumRGBWLeds;
127 pixels_device.mLenRemaining = mNumRGBWLeds;
128 uint8_t *data = reinterpret_cast<uint8_t *>(mRGBWPixels);
129 PixelIterator iterator = pixels.as_iterator(this->getRgbw());
130 while (iterator.has(1)) {
131 pixels.stepDithering();
132 iterator.loadAndScaleRGBW(data, data + 1, data + 2, data + 3);
133 data += 4;
134 iterator.advanceData();
135 }
136 // cast to base class to get around protected/private access issues
137 CPixelLEDController<RGB, LANES, MASK> &base = mController;
138 base.showPixels(pixels_device);
139 }
140
141 private:
142 // Needed by the interface.
143 void init() override {}
144
145 void ensureBuffer(int32_t num_leds) {
146 if (num_leds != mNumRGBLeds) {
147 mNumRGBLeds = num_leds;
148 // The delegate controller expects the raw pixel byte data in multiples of 3.
149 // In the case of src data not a multiple of 3, then we need to
150 // add pad bytes so that the delegate controller doesn't walk off the end
151 // of the array and invoke a buffer overflow panic.
152 mNumRGBWLeds = (num_leds * 4 + 2) / 3; // Round up to nearest multiple of 3
153 size_t extra = mNumRGBWLeds % 3 ? 1 : 0;
154 delete[] mRGBWPixels;
155 mRGBWPixels = new CRGB[mNumRGBWLeds + extra];
156 }
157 }
158
159 CRGB *mRGBWPixels = nullptr;
160 int32_t mNumRGBLeds = 0;
161 int32_t mNumRGBWLeds = 0;
162 CONTROLLER mController; // Real controller.
163};

312template <
313 uint8_t DATA_PIN, uint8_t CLOCK_PIN,
314 EOrder RGB_ORDER = RGB,
315 // APA102 has a bug where long strip can't handle full speed due to clock degredation.
316 // This only affects long strips, but then again if you have a short strip does 6 mhz actually slow
317 // you down? Probably not. And you can always bump it up for speed. Therefore we are prioritizing
318 // "just works" over "fastest possible" here.
319 // https://www.pjrc.com/why-apa102-leds-have-trouble-at-24-mhz/
320 uint32_t SPI_SPEED = DATA_RATE_MHZ(6),
321 FiveBitGammaCorrectionMode GAMMA_CORRECTION_MODE = kFiveBitGammaCorrectionMode_Null,
322 uint32_t START_FRAME = 0x00000000,
323 uint32_t END_FRAME = 0xFF000000
324>
325class APA102Controller : public CPixelLEDController<RGB_ORDER> {
326 typedef SPIOutput<DATA_PIN, CLOCK_PIN, SPI_SPEED> SPI;
327 SPI mSPI;
328
329 void startBoundary() {
330 mSPI.writeWord(START_FRAME >> 16);
331 mSPI.writeWord(START_FRAME & 0xFFFF);
332 }
333 void endBoundary(int nLeds) {
334 int nDWords = (nLeds/32);
335 const uint8_t b0 = uint8_t(END_FRAME >> 24 & 0x000000ff);
336 const uint8_t b1 = uint8_t(END_FRAME >> 16 & 0x000000ff);
337 const uint8_t b2 = uint8_t(END_FRAME >> 8 & 0x000000ff);
338 const uint8_t b3 = uint8_t(END_FRAME >> 0 & 0x000000ff);
339 do {
340 mSPI.writeByte(b0);
341 mSPI.writeByte(b1);
342 mSPI.writeByte(b2);
343 mSPI.writeByte(b3);
344 } while(nDWords--);
345 }
346
347 FASTLED_FORCE_INLINE void writeLed(uint8_t brightness, uint8_t b0, uint8_t b1, uint8_t b2) {
348#ifdef FASTLED_SPI_BYTE_ONLY
349 mSPI.writeByte(0xE0 | brightness);
350 mSPI.writeByte(b0);
351 mSPI.writeByte(b1);
352 mSPI.writeByte(b2);
353#else
354 uint16_t b = 0xE000 | (brightness << 8) | (uint16_t)b0;
355 mSPI.writeWord(b);
356 uint16_t w = b1 << 8;
357 w |= b2;
358 mSPI.writeWord(w);
359#endif
360 }
361
362 FASTLED_FORCE_INLINE void write2Bytes(uint8_t b1, uint8_t b2) {
363#ifdef FASTLED_SPI_BYTE_ONLY
364 mSPI.writeByte(b1);
365 mSPI.writeByte(b2);
366#else
367 mSPI.writeWord(uint16_t(b1) << 8 | b2);
368#endif
369 }
370
371public:
372 APA102Controller() {}
373
374 virtual void init() {
375 mSPI.init();
376 }
377
378protected:
380 virtual void showPixels(PixelController<RGB_ORDER> & pixels) {
381 PixelIterator iterator = pixels.as_iterator(this->getRgbw());
382 switch (GAMMA_CORRECTION_MODE) {
383 case kFiveBitGammaCorrectionMode_Null: {
384 showPixelsDefault(iterator);
385 break;
386 }
387 case kFiveBitGammaCorrectionMode_BitShift: {
388 showPixelsGammaBitShift(iterator);
389 break;
390 }
391 }
392 }

396 static inline void getGlobalBrightnessAndScalingFactors(
397 PixelIterator& pixels,
398 uint8_t* out_s0, uint8_t* out_s1, uint8_t* out_s2, uint8_t* out_brightness) {
399#if FASTLED_HD_COLOR_MIXING
400 uint8_t brightness;
401 pixels.getHdScale(out_s0, out_s1, out_s2, &brightness);
402 struct Math {
403 static uint16_t map(uint16_t x, uint16_t in_min, uint16_t in_max, uint16_t out_min, uint16_t out_max) {
404 const uint16_t run = in_max - in_min;
405 const uint16_t rise = out_max - out_min;
406 const uint16_t delta = x - in_min;
407 return (delta * rise) / run + out_min;
408 }
409 };
410 *out_brightness = Math::map(brightness, 0, 255, 0, 31);
411 return;
412#else
413 uint8_t s0, s1, s2;
414 pixels.loadAndScaleRGB(&s0, &s1, &s2);
415#if FASTLED_USE_GLOBAL_BRIGHTNESS == 1
416 // This function is pure magic.
417 const uint16_t maxBrightness = 0x1F;
418 uint16_t brightness = ((((uint16_t)max(max(s0, s1), s2) + 1) * maxBrightness - 1) >> 8) + 1;
419 s0 = (maxBrightness * s0 + (brightness >> 1)) / brightness;
420 s1 = (maxBrightness * s1 + (brightness >> 1)) / brightness;
421 s2 = (maxBrightness * s2 + (brightness >> 1)) / brightness;
422#else
423 const uint8_t brightness = 0x1F;
424#endif // FASTLED_USE_GLOBAL_BRIGHTNESS
425 *out_s0 = s0;
426 *out_s1 = s1;
427 *out_s2 = s2;
428 *out_brightness = static_cast<uint8_t>(brightness);
429#endif // FASTLED_HD_COLOR_MIXING
430 }
431
432 // Legacy showPixels implementation.
433 inline void showPixelsDefault(PixelIterator& pixels) {
434 mSPI.select();
435 uint8_t s0, s1, s2, global_brightness;
436 getGlobalBrightnessAndScalingFactors(pixels, &s0, &s1, &s2, &global_brightness);
437 startBoundary();
438 while (pixels.has(1)) {
439 uint8_t c0, c1, c2;
440 pixels.loadAndScaleRGB(&c0, &c1, &c2);
441 writeLed(global_brightness, c0, c1, c2);
442 pixels.stepDithering();
443 pixels.advanceData();
444 }
445 endBoundary(pixels.size());
446
447 mSPI.waitFully();
448 mSPI.release();
449 }

451 inline void showPixelsGammaBitShift(PixelIterator& pixels) {
452 mSPI.select();
453 startBoundary();
454 while (pixels.has(1)) {
455 // Load raw uncorrected r,g,b values.
456 uint8_t brightness, c0, c1, c2; // c0-c2 is the RGB data re-ordered for pixel
457 pixels.loadAndScale_APA102_HD(&c0, &c1, &c2, &brightness);
458 writeLed(brightness, c0, c1, c2);
459 pixels.stepDithering();
460 pixels.advanceData();
461 }
462 endBoundary(pixels.size());
463 mSPI.waitFully();
464 mSPI.release();
465 }
