243void CFastLED::countFPS(int nFrames) {
244 static int br = 0;
245 static uint32_t lastframe = 0; // millis();
246
247 if(br++ >= nFrames) {
248 uint32_t now = millis();
249 now -= lastframe;
250 if(now == 0) {
251 now = 1; // prevent division by zero below
252 }
253 m_nFPS = (br * 1000) / now;
254 br = 0;
255 lastframe = millis();
256 }
257}

259void CFastLED::setMaxRefreshRate(uint16_t refresh, bool constrain) {
260 if(constrain) {
261 // if we're constraining, the new value of m_nMinMicros _must_ be higher than previously (because we're only
262 // allowed to slow things down if constraining)
263 if(refresh > 0) {

266 } else if(refresh > 0) {

269 m_nMinMicros = 0;
