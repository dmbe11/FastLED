225 void init_binary_dithering() {
226#if !defined(NO_DITHERING) || (NO_DITHERING != 1)
227 // R is the digther signal 'counter'.
228 static uint8_t R = 0;
229 ++R;
230
231 // R is wrapped around at 2^ditherBits,
232 // so if ditherBits is 2, R will cycle through (0,1,2,3)
233 uint8_t ditherBits = VIRTUAL_BITS;
234 R &= (0x01 << ditherBits) - 1;
235
236 // Q is the "unscaled dither signal" itself.
237 // It's initialized to the reversed bits of R.
238 // If 'ditherBits' is 2, Q here will cycle through (0,128,64,192)
239 uint8_t Q = 0;
240
241 // Reverse bits in a byte
242 {
243 if(R & 0x01) { Q |= 0x80; }
244 if(R & 0x02) { Q |= 0x40; }
245 if(R & 0x04) { Q |= 0x20; }
246 if(R & 0x08) { Q |= 0x10; }
247 if(R & 0x10) { Q |= 0x08; }
248 if(R & 0x20) { Q |= 0x04; }
249 if(R & 0x40) { Q |= 0x02; }
250 if(R & 0x80) { Q |= 0x01; }
251 }
252
253 // Now we adjust Q to fall in the center of each range,
254 // instead of at the start of the range.
255 // If ditherBits is 2, Q will be (0, 128, 64, 192) at first,
256 // and this adjustment makes it (31, 159, 95, 223).
257 if( ditherBits < 8) {
258 Q += 0x01 << (7 - ditherBits);
259 }
260
261 // D and E form the "scaled dither signal"
262 // which is added to pixel values to affect the
263 // actual dithering.
264
265 // Setup the initial D and E values
266 for(int i = 0; i < 3; ++i) {
267 uint8_t s = mColorAdjustment.premixed.raw[i];
268 e[i] = s ? (256/s) + 1 : 0;
269 d[i] = scale8(Q, e[i]);
270#if (FASTLED_SCALE8_FIXED == 1)
271 if(d[i]) (--d[i]);
272#endif
273 if(e[i]) --e[i];
274 }
275#endif
276 }

518 FASTLED_FORCE_INLINE void loadAndScaleRGBW(Rgbw rgbw, uint8_t *b0_out, uint8_t *b1_out,
519 uint8_t *b2_out, uint8_t *b3_out) {
520#ifdef __AVR__
521 // Don't do RGBW conversion for AVR, just set the W pixel to black.
522 uint8_t out[4] = {
523 // Get the pixels in native order.
524 loadAndScale0(),
525 loadAndScale1(),
526 loadAndScale2(),
527 0,
528 };
529 EOrderW w_placement = rgbw.w_placement;
530 // Apply w-component insertion.
531 rgbw_partial_reorder(
532 w_placement, out[0], out[1], out[2],
533 0, // Pre-ordered RGB data with a 0 white component.
534 b0_out, b1_out, b2_out, b3_out);
535#else
536 const uint8_t b0_index = RGB_BYTE0(RGB_ORDER); // Needed to re-order RGB back into led native order.
537 const uint8_t b1_index = RGB_BYTE1(RGB_ORDER);
538 const uint8_t b2_index = RGB_BYTE2(RGB_ORDER);
539 // Get the naive RGB data order in r,g,b order.
540 CRGB rgb(mData[0], mData[1], mData[2]);
541 uint8_t w = 0;
542 rgb_2_rgbw(rgbw.rgbw_mode,
543 rgbw.white_color_temp,
544 rgb.r, rgb.b, rgb.g, // Input colors
545 mColorAdjustment.premixed.r, mColorAdjustment.premixed.g, mColorAdjustment.premixed.b, // How these colors are scaled for color balance.
546 &rgb.r, &rgb.g, &rgb.b, &w);
547 // Now finish the ordering so that the output is in the native led order for all of RGBW.
548 rgbw_partial_reorder(
549 rgbw.w_placement,
550 rgb.raw[b0_index], // in-place re-ordering for the RGB data.
551 rgb.raw[b1_index],
552 rgb.raw[b2_index],
553 w, // The white component is not ordered in this call.
554 b0_out, b1_out, b2_out, b3_out); // RGBW data now in total native led order.
555#endif
556 }
